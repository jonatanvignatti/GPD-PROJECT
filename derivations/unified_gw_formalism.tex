\section{Unified Stress-Energy Tensor for Gravitational Wave Sources}

We consider the cosmological first-order phase transition in the early universe, where gravitational waves (GWs) are produced through three primary mechanisms: bubble collisions, sound waves in the plasma, and turbulence. We adopt the convention of an expanding FLRW background with metric signature $(-+++)$.

\subsection{Combined Stress-Energy Tensor}

Assuming the mechanisms occur in distinct regimes or their cross-terms are negligible (linear superposition), the total stress-energy tensor $T_{\mu\nu}$ is given by:
\begin{equation}
    T_{\mu\nu} = T_{\mu\nu}^{\text{bub}} + T_{\mu\nu}^{\text{sw}} + T_{\mu\nu}^{\text{turb}}
\end{equation}

\subsubsection{Scalar Field (Bubbles)}
In the envelope approximation, the scalar field $\phi$ contributes to the stress-energy tensor primarily at the bubble walls. The components of the energy-momentum tensor are:
\begin{equation}
    T_{\mu\nu}^{\text{bub}} = \partial_\mu \phi \partial_\nu \phi - g_{\mu\nu} \left( \frac{1}{2} (\partial \phi)^2 + V(\phi) \right)
\end{equation}
The dominant source for GWs is the anisotropic stress $\Pi_{ij}^{\text{bub}}$, which in the TT gauge is determined by the gradients:
\begin{equation}
    \Pi_{ij}^{\text{bub}} \sim \partial_i \phi \partial_j \phi
\end{equation}

\subsubsection{Sound Waves (Acoustic Phase)}
Sound waves are bulk fluid motions in the plasma characterized by the enthalpy density $w = \rho + p$ and the velocity field $v^i$. The stress-energy tensor is:
\begin{equation}
    T_{\mu\nu}^{\text{sw}} = w U_\mu U_\nu + p g_{\mu\nu}
\end{equation}
where $U_\mu = (1, \mathbf{v})$ is the four-velocity of the fluid. In the non-relativistic limit for the bulk motions, the anisotropic stress is:
\begin{equation}
    \Pi_{ij}^{\text{sw}} = w v_i v_j
\end{equation}

\subsubsection{Turbulence}
Magnetohydrodynamic (MHD) turbulence at the end of the sound-wave phase contributes:
\begin{equation}
    T_{ij}^{\text{turb}} \approx w v_i^{\text{turb}} v_j^{\text{turb}}
\end{equation}

\subsection{Gravitational Wave Evolution Equation}

In an expanding FLRW background, the metric is given by $ds^2 = a^2(\eta) [-d\eta^2 + (\delta_{ij} + h_{ij}) dx^i dx^j]$. The linearized Einstein equation in the transverse-traceless (TT) gauge ($h_{0\mu} = 0, \partial^i h_{ij} = 0, h^i_i = 0$) is:
\begin{equation}
    h''_{ij} + 2\mathcal{H} h'_{ij} - \nabla^2 h_{ij} = 16\pi G a^2 \Pi_{ij}^{\text{TT}}
\end{equation}
where $\mathcal{H} = a'/a$ is the conformal Hubble rate, and $\Pi_{ij}^{\text{TT}}$ is the transverse-traceless projection of the combined anisotropic stress $\Pi_{ij} = T_{ij} - p g_{ij}$.

Under the assumption of linear superposition, the total source power spectrum is the sum of individual spectra:
\begin{equation}
    \langle \Pi_{ij}^{\text{TT}} \Pi_{ij}^{\text{TT}} \rangle = \langle \Pi_{ij, \text{bub}}^{\text{TT}} \Pi_{ij, \text{bub}}^{\text{TT}} \rangle + \langle \Pi_{ij, \text{sw}}^{\text{TT}} \Pi_{ij, \text{sw}}^{\text{TT}} \rangle + \langle \Pi_{ij, \text{turb}}^{\text{TT}} \Pi_{ij, \text{turb}}^{\text{TT}} \rangle
\end{equation}

\subsection{Gravitational Wave Spectral Density}

The energy density of the GWs is given by $\rho_{\text{GW}} = \frac{\langle \dot{h}_{ij} \dot{h}_{ij} \rangle}{32\pi G}$. The normalized spectral density is defined as:
\begin{equation}
    \Omega_{\text{GW}}(k) = \frac{1}{\rho_c} \frac{d \rho_{\text{GW}}}{d \ln k}
\end{equation}
where $\rho_c$ is the critical energy density of the universe. Assuming linear superposition of the three sources, the total spectrum is the sum:
\begin{equation}
    \Omega_{\text{GW}}(k) = \Omega_{\text{bub}}(k) + \Omega_{\text{sw}}(k) + \Omega_{\text{turb}}(k)
\end{equation}

\subsubsection{Bubble Collision Contribution (Envelope Approximation)}
The bubble collision spectrum, dominant in vacuum transitions, is characterized by the peak frequency $f_{\text{bub}} \sim \beta/v_w$ and efficiency $\kappa_{\text{bub}}$:
\begin{equation}
    \Omega_{\text{bub}}(k) \approx \left( \frac{H_*}{\beta} \right)^2 \left( \frac{\kappa_{\text{bub}} \alpha}{1 + \alpha} \right)^2 \left( \frac{v_w^3}{0.42 + v_w^2} \right) \frac{3.8 (k/k_{\text{bub}})^{2.8}}{1 + 2.8 (k/k_{\text{bub}})^{3.8}}
\end{equation}

\subsubsection{Sound Wave Contribution (Sound Shell Model)}
The sound wave contribution typically dominates in thermal transitions. Following Hindmarsh et al. (2015, 2019), the spectrum is:
\begin{equation}
    \Omega_{\text{sw}}(k) \approx 3 (H_* R_*) \left( \frac{\kappa_{\text{sw}} \alpha}{1 + \alpha} \right)^2 \tilde{\Omega}_{\text{sw}}(k) \Upsilon(t)
\end{equation}
where $R_* \approx (8\pi)^{1/3} v_w / \beta$ is the mean bubble separation, $\tilde{\Omega}_{\text{sw}}(k)$ is the spectral shape function, and $\Upsilon(t)$ is the suppression factor due to the finite lifetime of the acoustic phase:
\begin{equation}
    \Upsilon \approx \min(1, H_* \tau_{\text{sw}}), \quad \tau_{\text{sw}} \approx \frac{R_*}{U_f}
\end{equation}
The spectral shape $\tilde{\Omega}_{\text{sw}}(k)$ follows a double-broken power law:
\begin{equation}
    \tilde{\Omega}_{\text{sw}}(k) = \frac{\Omega_0}{(k/k_p)^{-3} + (k/k_p) \dots} \quad \text{(approximate)}
\end{equation}
A more precise fitting formula from Hindmarsh (2019) is used in numerical implementations:
\begin{equation}
    \tilde{\Omega}_{\text{sw}}(k) \approx (k/k_p)^3 \left( \frac{7}{4 + 3 (k/k_p)^2} \right)^{7/2}
\end{equation}

\subsubsection{Turbulence Contribution}
The turbulence contribution at late times (after $t > \tau_{\text{sw}}$) is modeled as:
\begin{equation}
    \Omega_{\text{turb}}(k) \approx 3 (H_* R_*) \left( \frac{\kappa_{\text{turb}} \alpha}{1 + \alpha} \right)^{3/2} \frac{(k/k_{\text{turb}})^3}{[1 + (k/k_{\text{turb}})]^{11/3} (1 + 8\pi k / H_*)}
\end{equation}
where the $k^{-11/3}$ factor corresponds to the Kolmogorov spectrum.

\subsection{Unified Formula}

The unified formula for the GW spectral density is:
\begin{equation}
    \Omega_{\text{GW}}(k) = \sum_{i \in \{\text{bub, sw, turb}\}} \mathcal{A}_i(v_w, \alpha, \beta/H_*) \mathcal{S}_i(k/k_i) \mathcal{T}_i(H_* \tau_i)
\end{equation}
where $\mathcal{A}_i$ are the peak amplitudes, $\mathcal{S}_i$ are the spectral shape functions, and $\mathcal{T}_i$ are the timescale-dependent suppression factors.

\subsubsection{IR Causality and UV Scaling}
As $k \to 0$, causality requires that the power spectrum of the energy density $P_{\rho}(k) \sim k^0$, which implies $d\rho/dk \sim k^2$, or $d\rho/d\ln k \sim k^3$. Thus, $\Omega_{\text{GW}}(k) \sim k^3$ in the IR limit for all components. In the UV, the spectra scale as $k^{-p}$ with $p$ depending on the mechanism (e.g., $p=1$ for bubbles in the envelope approximation, $p=3$ or $4$ for sound waves, $p=11/3$ for turbulence).

\section{Verification of Limiting Cases}

To verify the robustness of the unified formula, we examine it in well-known limiting cases and compare it with established benchmarks.

\subsection{Acoustic Limit (Sound Shell Model)}
In transitions where the latent heat is efficiently transferred to the plasma bulk motion and bubble collision energy is negligible ($\kappa_{\text{bub}} \to 0, \kappa_{\text{turb}} \to 0$), the unified formula reduces to:
\begin{equation}
    \Omega_{\text{GW}}(k) \to \Omega_{\text{sw}}(k) \approx 3 (H_* R_*) \left( \frac{\kappa_{\text{sw}} \alpha}{1 + \alpha} \right)^2 \tilde{\Omega}_{\text{sw}}(k) \Upsilon(t)
\end{equation}
This matches Equation (34) of Hindmarsh et al. (2015), where the GW spectral density is proportional to the mean bubble separation $R_*$ and the square of the efficiency factor $\kappa_{\text{sw}}$. The suppression factor $\Upsilon \approx \min(1, H_* \tau_{\text{sw}})$ accounts for the finite lifetime of the source, as discussed in Hindmarsh et al. (2017).

\subsection{Bubble Limit (Envelope Approximation)}
In vacuum phase transitions or cases where the plasma effect is negligible ($\kappa_{\text{sw}} \to 0, \kappa_{\text{turb}} \to 0$), the spectrum is dominated by the bubble wall collisions:
\begin{equation}
    \Omega_{\text{GW}}(k) \to \Omega_{\text{bub}}(k) \approx \left( \frac{H_*}{\beta} \right)^2 \left( \frac{\kappa_{\text{bub}} \alpha}{1 + \alpha} \right)^2 \left( \frac{v_w^3}{0.42 + v_w^2} \right) \mathcal{S}_{\text{bub}}(k/k_{\text{bub}})
\end{equation}
This reproduces the results from the envelope approximation (Huber \& Konstandin 2008, Caprini et al. 2016), where the peak frequency $f_{\text{bub}}$ scales with $\beta/v_w$ and the amplitude scales with $(H_*/\beta)^2$.

\subsection{Turbulence Limit and UV Scaling}
At late times, after the sound wave period ($\tau > \tau_{\text{sw}}$), the turbulence contribution dominates. The unified formula specifies a UV scaling:
\begin{equation}
    \Omega_{\text{turb}}(k) \propto k^{-11/3} \quad \text{for } k \gg k_{\text{turb}}
\end{equation}
which is consistent with the Kolmogorov spectrum for stationary turbulence. In the extreme UV, additional suppression factors like $(1 + 8\pi k / H_*)$ ensure the energy density remains finite.

\subsection{Combined Peak Shift}
The peak frequency of the combined spectrum is determined by the dominant source. For most thermal transitions, the sound wave peak $k_{\text{sw}} \sim 2/R_*$ dominates. Since $R_* \approx (8\pi)^{1/3} v_w / \beta$, the peak frequency shifts to lower values as the wall velocity $v_w$ increases (for a fixed $\beta$), or to higher frequencies as the transition rate $\beta$ increases. This trend is consistent with all benchmark simulations.
